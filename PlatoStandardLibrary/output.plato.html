Library Array
{
    Map(xs : Array, f : Function) : Array => 
        xs.Count.Tuple((i) => 
            (xs [] i).f);

    Reverse(xs : Array) : Array => 
        xs.Count.Tuple((i) => 
            (xs [] (xs.Count - (1 - i))).f);

    Zip(xs : Array, ys : Array, f : Function) : Array => 
        xs.Count.Tuple((i) => 
            i.At.f((ys [] i)));

    Zip(xs : Array, ys : Array, zs : Array, f : Function) : Array => 
        xs.Count.Tuple((i) => 
            i.At.f((ys [] i), (zs [] i)));

    Skip(xs : Array, n : Count) : Array => 
        (Count - n).Tuple((i) => 
            (i - n).At);

    Take(xs : Array, n : Count) : Array => 
        n.Tuple((i) => 
            i.At);

    Aggregate(xs : Array, init : Any, f : Function) : Any => 
        xs.IsEmpty
            ? init
            : init.f(xs.Rest.f);

    Rest(xs : Array) : Array => 
        xs.Skip(1);

    IsEmpty(xs : Array) : Boolean => 
        (xs.Count == 0);

    First(xs : Array) : Any => 
        (xs [] 0);

    Last(xs : Array) : Any => 
        (xs [] (xs.Count - 1));

    Slice(xs : Array, from : Index, count : Count) : Array => 
        xs.Skip(from).Take(count);

    Join(xs : Array, sep : String) : String => 
        xs.IsEmpty
            ? 
            : (xs.First.ToString + xs.Rest.Aggregate(, (acc, cur) => 
                acc.Interpolate(sep, cur)));

    All(xs : Array, f : Function) : Boolean => 
        xs.IsEmpty
            ? True
            : (xs.First.f && xs.Rest.f);

}
Library Interval
{
    Size(x : Interval) : Numerical => 
        (x.Max - x.Min);

    IsEmpty(x : Interval) : Boolean => 
        (x.Min >= x.Max);

    Lerp(x : Interval, amount : Unit) : Numerical => 
        (x.Min * ((1 - amount) + (x.Max * amount)));

    InverseLerp(x : Interval, value : Numerical) : Unit => 
        ((value - x.Min) / x.Size);

    Negate(x : Interval) : Interval => 
        -x.Max.Tuple(-x.Min);

    Reverse(x : Interval) : Interval => 
        x.Max.Tuple(x.Min);

    Center(x : Interval) : Numerical => 
        x.Lerp(0.5);

    Contains(x : Interval, value : Numerical) : Boolean => 
        (x.Min <= (value && (value <= x.Max)));

    Contains(x : Interval, other : Interval) : Boolean => 
        (x.Min <= (other.Min && (Max >= other.Max)));

    Overlaps(x : Interval, y : Interval) : Boolean => 
        !x.Clamp(y).IsEmpty;

    Split(x : Interval, t : Unit) : Tuple => 
        x.Left(t).Tuple(x.Right(t));

    Split(x : Interval) : Tuple => 
        x.Split(0.5);

    Left(x : Interval, t : Unit) : Interval => 
        x.Min.Tuple(x.Lerp(t));

    Right(x : Interval, t : Unit) : Interval => 
        x.Lerp(t).Tuple(x.Max);

    MoveTo(x : Interval, v : Numerical) : Interval => 
        v.Tuple((v + x.Size));

    LeftHalf(x : Interval) : Interval => 
        x.Left(0.5);

    RightHalf(x : Interval) : Interval => 
        x.Right(0.5);

    HalfSize(x : Interval) : Numerical => 
        x.Size.Half;

    Recenter(x : Interval, c : Numerical) : Interval => 
        (c - x.HalfSize).Tuple((c + x.HalfSize));

    Clamp(x : Interval, y : Interval) : Interval => 
        x.Clamp(y.Min).Tuple(x.Clamp(y.Max));

    Clamp(x : Interval, value : Numerical) : Numerical => 
        (value < x.Min
            ? x.Min
            : (value > x.Max
                ? x.Max
                : value));

    Within(x : Interval, value : Numerical) : Boolean => 
        (value >= (x.Min && (value <= x.Max)));

}

Library Angles
{
    Radians(x : Number) : Angle => 
        x;

    Degrees(x : Number) : Angle => 
        (x * (Pi / 180));

    Turns(x : Number) : Angle => 
        (x * (2 * Pi));

}
Library Comparable
{
    Equals(a : Comparable, b : Comparable) : Boolean => 
        (a.Compare(b) == 0);

    LessThan(a : Comparable, b : Comparable) : Boolean => 
        (a.Compare(b) < 0);

    LessThanOrEquals(a : Comparable, b : Comparable) : Boolean => 
        (a.Compare(b) <= 0);

    GreaterThan(a : Comparable, b : Comparable) : Boolean => 
        (a.Compare(b) > 0);

    GreaterThanOrEquals(a : Comparable, b : Comparable) : Boolean => 
        (a.Compare(b) >= 0);

    Between(v : Comparable, a : Comparable, b : Comparable) : Value => 
        (v >= (a && (v <= b)));

    Between(v : Value, i : Interval) : Interval => 
        i.Contains(v);

    Min(a : Comparable, b : Comparable) : Comparable => 
        (a <= b
            ? a
            : b);

    Max(a : Comparable, b : Comparable) : Comparable => 
        (a >= b
            ? a
            : b);

}
Library Equatable
{
    NotEquals(x : Equatable, y : Equatable) : Boolean => 
        !(x == y);

}
Library Easings
{
    BlendEaseFunc(p : Number, easeIn : Function, easeOut : Function) : Number => 
        (p < 0.5
            ? (0.5 * (p * 2).easeIn)
            : (0.5 * ((p * (2 - 1)).easeOut + 0.5)));

    InvertEaseFunc(p : Number, easeIn : Function) : Number => 
        (1 - (1 - p).easeIn);

    Linear(p : Number) : Number => 
        p;

    QuadraticEaseIn(p : Number) : Number => 
        p.Pow2;

    QuadraticEaseOut(p : Number) : Number => 
        p.InvertEaseFunc(QuadraticEaseIn);

    QuadraticEaseInOut(p : Number) : Number => 
        p.BlendEaseFunc(QuadraticEaseIn, QuadraticEaseOut);

    CubicEaseIn(p : Number) : Number => 
        p.Pow3;

    CubicEaseOut(p : Number) : Number => 
        p.InvertEaseFunc(CubicEaseIn);

    CubicEaseInOut(p : Number) : Number => 
        p.BlendEaseFunc(CubicEaseIn, CubicEaseOut);

    QuarticEaseIn(p : Number) : Number => 
        p.Pow4;

    QuarticEaseOut(p : Number) : Number => 
        p.InvertEaseFunc(QuarticEaseIn);

    QuarticEaseInOut(p : Number) : Number => 
        p.BlendEaseFunc(QuarticEaseIn, QuarticEaseOut);

    QuinticEaseIn(p : Number) : Number => 
        p.Pow5;

    QuinticEaseOut(p : Number) : Number => 
        p.InvertEaseFunc(QuinticEaseIn);

    QuinticEaseInOut(p : Number) : Number => 
        p.BlendEaseFunc(QuinticEaseIn, QuinticEaseOut);

    SineEaseIn(p : Number) : Number => 
        p.InvertEaseFunc(SineEaseOut);

    SineEaseOut(p : Number) : Number => 
        p.Quarter.Turns.Sin;

    SineEaseInOut(p : Number) : Number => 
        p.BlendEaseFunc(SineEaseIn, SineEaseOut);

    CircularEaseIn(p : Number) : Number => 
        p.Pow2.FromOne.SquareRoot.FromOne;

    CircularEaseOut(p : Number) : Number => 
        p.InvertEaseFunc(CircularEaseIn);

    CircularEaseInOut(p : Number) : Number => 
        p.BlendEaseFunc(CircularEaseIn, CircularEaseOut);

    ExponentialEaseIn(p : Number) : Number => 
        p.AlmostZero
            ? p
            : 2.Pow((10 * p.MinusOne));

    ExponentialEaseOut(p : Number) : Number => 
        p.InvertEaseFunc(ExponentialEaseIn);

    ExponentialEaseInOut(p : Number) : Number => 
        p.BlendEaseFunc(ExponentialEaseIn, ExponentialEaseOut);

    ElasticEaseIn(p : Number) : Number => 
        (13 * (p.Quarter.Turns * 2.Pow((10 * p.MinusOne)).Radians.Sin));

    ElasticEaseOut(p : Number) : Number => 
        p.InvertEaseFunc(ElasticEaseIn);

    ElasticEaseInOut(p : Number) : Number => 
        p.BlendEaseFunc(ElasticEaseIn, ElasticEaseOut);

    BackEaseIn(p : Number) : Number => 
        (p.Pow3 - (p * p.Half.Turns.Sin));

    BackEaseOut(p : Number) : Number => 
        p.InvertEaseFunc(BackEaseIn);

    BackEaseInOut(p : Number) : Number => 
        p.BlendEaseFunc(BackEaseIn, BackEaseOut);

    BounceEaseIn(p : Number) : Number => 
        p.InvertEaseFunc(BounceEaseOut);

    BounceEaseOut(p : Number) : Number => 
        (p < (4 / 11))
            ? (121 * (p.Pow2 / 16))
            : (p < (8 / 11))
                ? (363 / (40 * (p.Pow2 - (99 / (10 * (p + (17 / 5)))))))
                : (p < (9 / 10))
                    ? (4356 / (361 * (p.Pow2 - (35442 / (1805 * (p + (16061 / 1805)))))))
                    : (54 / (5 * (p.Pow2 - (513 / (25 * (p + (268 / 25)))))));

    BounceEaseInOut(p : Number) : Number => 
        p.BlendEaseFunc(BounceEaseIn, BounceEaseOut);

}
